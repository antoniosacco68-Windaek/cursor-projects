document.addEventListener('DOMContentLoaded', function() {
    const tireData = tireDataString.trim().split('\n').map(line => line.trim()).filter(line => line.length > 0);
    const tiresObj = {};
    
    tireData.forEach(tireStr => {
        const [larghezza, sezione, diametro, carico, velocita] = tireStr.split('-');
        if (!tiresObj[larghezza]) tiresObj[larghezza] = {};
        if (!tiresObj[larghezza][sezione]) tiresObj[larghezza][sezione] = {};
        if (!tiresObj[larghezza][sezione][diametro]) tiresObj[larghezza][sezione][diametro] = {};
        if (!tiresObj[larghezza][sezione][diametro][carico]) tiresObj[larghezza][sezione][diametro][carico] = [];
        tiresObj[larghezza][sezione][diametro][carico].push(velocita);
    });

    function updateSelectors() {
        const larghezzaSelect = document.getElementById('larghezza-pneumatico');
        const sezioneSelect = document.getElementById('sezione-pneumatico');
        const diametroSelect = document.getElementById('diametro-pneumatico');
        const caricoSelect = document.getElementById('carico-pneumatico');
        const velocitaSelect = document.getElementById('velocita-pneumatico');
        
        const larghezzaValue = larghezzaSelect.value;
        const sezioneValue = sezioneSelect.value;
        const diametroValue = diametroSelect.value;
        const caricoValue = caricoSelect.value;
        
        // Reset and update sezione based on larghezza
        if (larghezzaValue && larghezzaValue !== '- - Larghezza - -') {
            const availableSezioni = Object.keys(tiresObj[larghezzaValue] || {});
            sezioneSelect.innerHTML = '<option value="- - Spalla - -">- - Spalla - -</option>';
            availableSezioni.forEach(sez => {
                sezioneSelect.innerHTML += `<option value="${sez}">${sez}</option>`;
            });
            sezioneSelect.style.color = '#757575';
        }
        
        // Reset and update diametro based on larghezza and sezione
        if (larghezzaValue && sezioneValue && sezioneValue !== '- - Spalla - -') {
            const availableDiametri = Object.keys(tiresObj[larghezzaValue][sezioneValue] || {});
            diametroSelect.innerHTML = '<option value="- - Diametro - -">- - Diametro - -</option>';
            availableDiametri.forEach(diam => {
                diametroSelect.innerHTML += `<option value="${diam}">${diam}</option>`;
            });
            diametroSelect.style.color = '#757575';
        }
        
        // Reset and update carico based on larghezza, sezione, and diametro
        if (larghezzaValue && sezioneValue && diametroValue && diametroValue !== '- - Diametro - -') {
            const availableCarichi = Object.keys(tiresObj[larghezzaValue][sezioneValue][diametroValue] || {});
            caricoSelect.innerHTML = '<option value="- - Carico - -">- - Carico - -</option>';
            availableCarichi.forEach(car => {
                caricoSelect.innerHTML += `<option value="${car}">${car}</option>`;
            });
            caricoSelect.style.color = '#757575';
        }
        
        // Reset and update velocita based on larghezza, sezione, diametro, and carico
        if (larghezzaValue && sezioneValue && diametroValue && caricoValue && caricoValue !== '- - Carico - -') {
            const availableVelocita = tiresObj[larghezzaValue][sezioneValue][diametroValue][caricoValue] || [];
            velocitaSelect.innerHTML = '<option value="- - Velocità - -">- - Velocità - -</option>';
            availableVelocita.forEach(vel => {
                velocitaSelect.innerHTML += `<option value="${vel}">${vel}</option>`;
            });
            velocitaSelect.style.color = '#757575';
        }
        
        // Aggiorna i colori dei select
        fixPlaceholderColors();
    }
    
    const selectors = document.querySelectorAll('.tire-select');
    selectors.forEach(selector => {
        selector.addEventListener('change', updateSelectors);
    });
    
    // Funzione per sistemare i colori dei placeholder
    function fixPlaceholderColors() {
        // Trova tutti i select nel form
        const selects = document.querySelectorAll('select');
        
        selects.forEach(select => {
            // Colore iniziale del testo in base al valore selezionato
            if (select.selectedIndex === 0) {
                select.style.color = '#757575'; // Grigio per l'opzione placeholder
            } else {
                select.style.color = '#000000'; // Nero per qualsiasi altra opzione
            }
            
            // Ascoltatore per il cambio di valore
            select.addEventListener('change', function() {
                if (this.selectedIndex === 0) {
                    this.style.color = '#757575'; // Grigio per l'opzione placeholder
                } else {
                    this.style.color = '#000000'; // Nero per qualsiasi altra opzione
                }
            });
        });
    }
    
    // Esegui la funzione dopo che il DOM è completamente caricato
    fixPlaceholderColors();
    
    // Esegui la funzione anche dopo un breve ritardo per assicurarsi che 
    // Contact Form 7 abbia finito di costruire completamente il form
    setTimeout(fixPlaceholderColors, 500);
});
</script>
